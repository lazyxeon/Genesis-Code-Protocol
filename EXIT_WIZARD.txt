1. Validate schema and run tests
   make install && make test
2. Build container, generate SBOM, sign artifacts
   make package && make sign
3. Launch canary deployment with OTEL enabled
   docker run -e OTEL_EXPORTER_OTLP_ENDPOINT=$OTEL_ENDPOINT matrix-ci-repair --canary
4. Run 15-minute SLO probe; promote or rollback
   sleep 900 && curl -f $HEALTHCHECK || docker rollback matrix-ci-repair
5. Emit release receipt
   jq -n '{commit: "$GIT_COMMIT", image: "matrix-ci-repair", sbom: "$(sha256sum sbom.json | cut -d" " -f1)"}' > release_receipt.json
